Padding

CSS tem propriedades para especificar o preenchimento para cada lado de um elemento:
  • padding-top
  • padding-right
  • padding-bottom
  • padding-left
Todas as propriedades de preenchimento podem ter os seguintes valores:
  • length - especifica um preenchimento em px, pt, cm, etc.
  • % - especifica um preenchimento em % da largura do elemento contido.
  • inherit - especifica que o preenchimento deve ser herdado do elemento pai.
Nota: Valor negativo não é aceito.

Imagem ilustrativa:

  • Nomenclatura

Position

Existem cinco valores de posição diferentes:
  • static
    • Valor padrão. Elementos são renderizados em ordem, conforme aparecem no fluxo de documentos
  • relative
    • O elemento é posicionado em relação ao primeiro elemento ancestral posicionado (não estático)
  • fixed
    • O elemento está posicionado em relação à janela do navegador
  • absolute
    • O elemento está posicionado em relação à sua posição normal, então "left: 20px" adiciona 20 pixels à posição LEFT do elemento
  • sticky
    • O elemento é posicionado com base na posição de rolagem do usuário Um elemento adesivo alterna entre relativo e fixo, dependendo da posição de rolagem. Ela é posicionada relativa até que uma determinada posição de deslocamento seja atendida na viewport - então ela "gruda" no lugar (como position: fixed).

  • Estrutura
  • Nomenclatura

div {
    position: relative;
}

	chartRentabilidade : function(component) {
        console.log('entrou ');
        var canvas;
        var context;
        var Val_max;
        var Val_min;
        var sections;
        var xScale;
        var yScale;
        var ln = 0;
        var margFix = 40;
        var ln_color = new Array();
        
        ln_color[0] = "purple";
        var vetRentabilidade = ['v.rentabilidadeMes','v.rentabilidadeAno','v.rentabilidade12meses'];
        var vetCDI = ['v.cdiMes','v.cdiAno','v.cdi12meses'];
        var idCanvas = ['chartRentabilidadeMes','chartRentabilidadeAno','chartRentabilidade12meses'];
        
        var opRentabilidade;
        var opCDI;
        
        ln_color[1] = "#4bf087";
        ln_color[2] = "#e288d7";
        
        for(var i=0; i<3;i++ ){
            var id = idCanvas[i];
            canvas = document.getElementById(id);
            var rentabilidade = component.get(vetRentabilidade[i]);
            var cdi =  component.get(vetCDI[i]);
            init(canvas, rentabilidade, cdi );
        }
        function init(canvas, rentabilidade, cdi) {
            // set these values for your data
            sections = 9;
            Val_max = 100;
            Val_min = 0;
            var stepSize = 10;
            var columnSize = 250;
            var rowSize = 50;
            var margin = 10;
            var xAxis = [" ", "01", "02", "03", "04", "05", "06", "07", "08", "09"];
            
            
            
            
            context = canvas.getContext("2d");
            context.fillStyle = "#222";
            context.font = "14px Verdana";
            yScale = (canvas.height - columnSize - margin) / (Val_max - Val_min);
            xScale = (canvas.width - rowSize) / sections;
            
            context.strokeStyle = "#979797"; // color of grid lines
            context.lineWidth = 0.6;
            // horizontal row
            context.moveTo(canvas.width-margFix,canvas.height - margFix);
            context.lineTo((margFix-margin),canvas.height - margFix);
            context.setLineDash([1, 0]);
            context.stroke();
            
            context.setLineDash([5, 4]); //dashes are 5px and spaces are 3px
            context.beginPath();
            // print Parameters on X axis, and grid lines on the graph
            
            for (var i = 1; i <= sections; i++) {
                var x = i * xScale;
                context.fillText(xAxis[i], x - 10, canvas.height - margin);
                context.moveTo(x, columnSize - margFix);
                context.lineTo(x, canvas.height - margin - margFix);
            }
            
            // print row header and draw horizontal grid lines
            
            context.stroke();
            
            context.translate(rowSize, canvas.height + Val_min * yScale);
            context.scale(1, -1 * yScale);
            
            context.strokeStyle = ln_color[1];
            context.setLineDash([1, 0]);
            
            plotData(rentabilidade,1,canvas);
            context.strokeStyle = ln_color[2];
            
            plotData(cdi,2,canvas);
        }
        
        function plotData(dataSet, ln,canvas) {
            var circle_x;
            var circle_y;
            context.beginPath();
            context.moveTo(-10, dataSet[0] + margFix);
            for (var i = 1; i < sections; i++) {
                context.lineTo(i * xScale - 10, dataSet[i] + margFix);
                circle_x = i * xScale - 12;
                circle_y = dataSet[i] + 15;
            }
            context.stroke();
            
            //circle
            var ctx = canvas.getContext("2d");
            
            ctx.beginPath();
            ctx.arc(circle_x, (circle_y+margFix-15), 4, 0, 2 * Math.PI);
            ctx.strokeStyle = ln_color[ln];
            ctx.fillStyle = ln_color[ln];
            ctx.stroke();
            ctx.fill();
        }
        
    }
})
