function CorrigeData()

  • Arquivo: scripts.js
Corrige a data digitada e testifica se é valida.
Recebe a data, verifica se contém 8 digitos, não possui "/" e se corresponde a uma data;

Chamar função: <input type="text" class="corrigeData">;

  • JQuery
  • Parâmetros
  • Chamada
// alert("ready");
 $('.corrigeData').focusout(function (e) {
     var campo = $(this).val();
     if (campo.length == 8 && campo.indexOf('/') < 0) {
    
         if (((campo.substring(0, 2) == 30) && (campo.substring(2, 4) == 2)) ||
            ((campo.substring(0, 2) == 31) &&                 (campo.substring(2, 4) == 2))) {
             $(this).val("");
    
         } else {
             $(this).val(campo.substring(0, 2) + '/' + campo.substring(2, 4) + '/' + campo.substring(4, 8));
    
         }
     } else {
    
         if (campo.length == 10) {
             if ((campo.substring(0, 2) > 31) || (campo.substring(3, 5) > 12) ||
                (campo.substring(6, 10) < 1900)) 
             {
                 $(this).val("");
    
             }
             if (((campo.substring(0, 2) == 30) && (Conteudo.substring(3, 5) == 2)) ||
                ((campo.substring(0, 2) == 31) && (campo.substring(3, 5) == 2))) {
                 $(this).val("");
    
             }
             if (isDate($(this).val()) == false) {
                 $(this).val("");
    
             }
         } else {
             if (isDate($(this).val()) == false) {
                 $(this).val("");
             }
         }
     }
     e.preventDefault();
 });

function FormatDecimal()

  • Arquivo: scripts.js
Função que organiza os números com casas decimais inseridas no portal.
Recebe um número, verifica o dado recebido, caso possua mais de três casas inclui um ponto, organiza a quantidade de casas decimais e após o número insere dois zeros.

Ex:
Entrada: 2563
Resultado: 2.563,00

Chamar função: <input type="text" class="formatDecimal">;

  • JQuery
  • Parâmetros
  • Chamada
$('.formatDecimal').blur(function () {

    //pStrField, vIntQdDec, pStrSeparadorDecimal, pBlnIsRequired
    var vIntPosPonto, vIntPosVirgula;
    var vStrNewValue = pStrField.value;


    // remove todos os pontos digitados
    while (pStrField.value.indexOf(".") > 0)
        pStrField.value = pStrField.value.replace("\.", "");

    if ((pStrField.value.length > 0) || pBlnIsRequired) {
        if ((!check_decimal(pStrField.value, true)) || (pStrField.value == 0)) {
            if (pStrField.value == 0) {
                pStrField.value = pStrField.value;
            } else {
                pStrField.select();
            }
        } else {
            vIntPosVirgula = pStrField.value.indexOf(pStrSeparadorDecimal);
            if (vIntPosVirgula > -1) {
                if (vIntPosVirgula == 0)
                    vStrNewValue = "0," + pStrField.value.substring(vIntPosVirgula + 1, pStrField.value.length);
                else
                    vStrNewValue = pStrField.value.substring(0, vIntPosVirgula) + pStrSeparadorDecimal +
                    pStrField.value.substring(vIntPosVirgula + 1, pStrField.value.length);

                if (vStrNewValue.length - vIntPosVirgula < vIntQdDec + 1) {
                    for (var i = vStrNewValue.length - vIntPosVirgula; i < vIntQdDec + 1; i++)
                        vStrNewValue = vStrNewValue + "0";
                } else if (vStrNewValue.length - vIntPosVirgula > vIntQdDec + 1)
                    vStrNewValue = vStrNewValue.substring(0, vIntPosVirgula + vIntQdDec + 1);
            } else
                vStrNewValue = pStrField.value + pStrSeparadorDecimal + "00";

            var tempStr = "";
            var startCount = vStrNewValue.lastIndexOf(",") - 1;
            for (var x = startCount; x >= 0; x--) {
                if (((startCount - x) != 0) && ((startCount - x) % 3 == 0))
                    tempStr = "." + tempStr;
                tempStr = vStrNewValue.charAt(x) + tempStr;
            }
            pStrField.value = tempStr + vStrNewValue.substring(vStrNewValue.lastIndexOf(","), vStrNewValue.length);
        }
    }

    var varEmail = email.value;

    if (intPosCarac == -1) {
        return email.value = "";
    } else {
        var strDepois = varEmail.substr(intPosCarac + 1, varEmail.length - intPosCarac);
        var strAntes = varEmail.substr(0, intPosCarac);
        intPosCarac = strDepois.indexOf(".");
        
        if (strAntes.charAt(strAntes.length - 1) == ".") {
            email.value = "";

            return email.value = "";
